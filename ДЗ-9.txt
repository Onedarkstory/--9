using System;

namespace Exit
{
    internal class Program
    {
        static void Main(string[] args)
        {
            string usingInput = "";
            string exit = "exit";

            while (usingInput != exit)
            {
                Console.Write("Введите ваше сообщение (Для выхода из программы введите exit): ");
                usingInput = Console.ReadLine();

                Console.WriteLine($"Вы ввели: {usingInput} ");
            }
            Console.WriteLine("Программа завершена!");
        }
    }
}
